// All partitions
void bitmaskSplit(string s){
    int n = s.size();
    int ways = (1 << (n - 1));

    for (int mask = 0; mask < ways; mask++){
        string part;
        for (int i=0; i<n; i++){
            part += s[i];
            if (mask & (1 << i)){
                cout<<part<<" ";
                part.clear();
            }
        }
        cout<<part<<"\n";
    }
}
